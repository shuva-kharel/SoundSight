# SoundSight backend dependencies
fastapi
uvicorn[standard]
ultralytics
easyocr
opencv-python
websockets
python-multipart
cryptography    # generates the self-signed cert for HTTPS (browsers force it on localhost)
google-genai    # Describe mode: Google Gemini vision API (needs GEMINI_API_KEY)
gTTS            # nicer Nepali voice for ARBITRARY OCR text + to pre-generate the fixed
                # phrases (run `python gen_voice.py` once, online, -> audio/ne/*.mp3)
python-dotenv   # loads GEMINI_API_KEY etc. from a .env file in the project root
# Nepali speech is TIERED: pre-recorded audio/ne/ files (best, offline) for the fixed
# vocabulary -> gTTS (good, online) for arbitrary text -> espeak-ng (offline fallback).
# espeak-ng is a program, not a pip package: Windows `winget install eSpeak-NG.eSpeak-NG`,
# Pi/Linux `sudo apt install espeak-ng`. English speech is the browser's Web Speech API.
#
# Faces (enabled on the laptop): InsightFace buffalo_l via onnxruntime.
insightface     # known-face recognition (greet by name); pure-python wheel
onnxruntime     # runs the InsightFace model (CPU). For GPU: onnxruntime-gpu (matching CUDA)
# Note: MODEL_MODE="openvocab" (YOLO-World) also needs CLIP -- ultralytics
# auto-installs it on first use, so it isn't pinned here.

# NOTE on PyTorch / GPU:
# ultralytics pulls in `torch` automatically. The default CPU/cu12x wheels do
# NOT support newer Blackwell GPUs (RTX 50-series, sm_120). If YOLO falls back
# to CPU or errors on CUDA, install a matching torch build first, e.g.:
#   pip install --pre torch torchvision --index-url https://download.pytorch.org/whl/nightly/cu128
