# SoundSight -- Raspberry Pi 4 dependencies (64-bit Raspberry Pi OS, aarch64).
#
# *** 64-bit OS required. The Pi has NO GPU -- torch MUST be the CPU build. ***
# Modern torch defaults to CUDA aarch64 wheels (nvidia-*, triton) that waste
# gigabytes and don't work on a Pi. The --extra-index-url + ordering below force
# the CPU build. The app runs the light path on the Pi (NCNN nano, cpu, imgsz 320).
#
# Install with: bash setup_pi.sh

--extra-index-url https://download.pytorch.org/whl/cpu

torch                # CPU build, forced by the index URL above -- install BEFORE ultralytics
torchvision

ultralytics          # YOLO (torch already satisfied above, so it won't pull CUDA)
opencv-python-headless # camera + image ops without GUI/Qt libs (less crashy on headless Pi)
numpy
easyocr              # Read mode OCR (Nepali + English); downloads models on first use
vosk                 # offline speech recognition (download a model -- see README_PI.md)
sounddevice          # microphone capture for Vosk (apt needs portaudio19-dev)
piper-tts            # optional nicer offline TTS; espeak-ng (apt) is the default
gpiozero             # optional: physical GPIO button to trigger modes
python-dotenv        # loads GEMINI_API_KEY from .env (only if you want online Describe)
#   google-genai     # online Describe (needs internet + GEMINI_API_KEY)
#   gTTS             # online TTS
